# Meetio Colors

The most delicious and wonderful colors for Meetio Theme

### Installation

```bash
# NPM
$ npm install @meetio/meetio-colors -D

# YARN
$ yarn add @meetio/meetio-colors
```


### Interface

```ts
export interface IColors {
    cursor: string;
    foreground: string;
    background: string;
    comments: string;
    blue: string;
    brown: string;
    cyan: string;
    green: string;
    orange: string;
    pink: string;
    purple: string;
    red: string;
    yellow: string;
    invalid: string;
    deprecated: string;
    diffAdded: string;
    diffModified: string;
    diffDeleted: string;
}
